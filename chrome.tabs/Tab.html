Tab class 

class Tab {
 final int id;
 final int index;
 final int windowId;
 final int openerTabId;
 final bool highlighted;
 final bool active;
 final bool pinned;
 final String url;
 final String title;
 final String favIconUrl;
 final TabStatus status;
 final bool incognito;

 const Tab._(
     this.id,
     this.index,
     this.windowId,
     this.openerTabId,
     this.highlighted,
     this.active,
     this.pinned,
     this.url,
     this.title,
     this.favIconUrl,
     this.status,
     this.incognito);


 Tab(tab) : this._(
     tab.id,
     tab.index,
     tab.windowId,
     tab['openerTabId'],
     tab.highlighted,
     tab.active,
     tab.pinned,
     tab['url'],
     tab['title'],
     tab['favIconUrl'],
     tab['status'] != null ? new TabStatus(tab['status']) : null,
     tab.incognito);
}
