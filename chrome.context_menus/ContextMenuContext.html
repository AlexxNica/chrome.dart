ContextMenuContext class 

class ContextMenuContext {
 final String _context;

 const ContextMenuContext._internal(this._context);

 factory ContextMenuContext(String context) {
   switch (context.toLowerCase()) {
     case 'all':
       return ALL;
     case 'page':
       return PAGE;
     case 'frame':
       return FRAME;
     case 'selection':
       return SELECTION;
     case 'link':
       return LINK;
     case 'editable':
       return EDITABLE;
     case 'image':
       return IMAGE;
     case 'video':
       return VIDEO;
     case 'audio':
       return AUDIO;
     case 'launcher':
       return LAUNCHER;
     default:
       throw 'Unknown ContextMenuContext: $context';
   }
 }

 String toString() {
   return _context;
 }
