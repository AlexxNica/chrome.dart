   list.add(jsArray[i]);
 }
 // TODO(DrMarcII) consider having this return an unmodifiable list

dynamic jsifyMessage(message) #

dynamic jsifyMessage(dynamic message) {
 if (message is Map) {
   return js.map(message);
 } else if (message is Iterable) {
   return js.array(message);
 } else {
   return message;
 }
}
